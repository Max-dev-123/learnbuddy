          if (t.includes("add 9")) return "Yes. What is 14 + 9?";

          if (t.includes("23")) return "Nice. Now check: 23 − 9 = ?";

          if (t.includes("14")) return "Great check! Explain: what did you do to both sides first?";

          return "Tell me: what operation is being done to x (−9, +9, ×, or ÷)?";

        }

      },


      {

        id: "eq2",

        title: "Two-step Equations (Long Lesson)",

        difficulty: 3,

        time: "30–40 min",

        skills: "Undo two operations",

        goal: "Solve equations like 3x + 7 = 22 using reverse operations and checking.",

        steps: [

          "Circle the term with x (like 3x).",

          "Undo addition/subtraction first.",

          "Then undo multiplication/division.",

          "Write one line per step so it’s easy to follow.",

          "Check your answer by substituting back in.",

          "Try a second question to confirm you can repeat the process."

        ],

        practice: "Solve: 4x + 6 = 30",

        hints: [

          "What should you undo first: +6 or ×4?",

          "Undo +6 by subtracting 6 from BOTH sides.",

          "30 − 6 = 24, so now you have 4x = 24.",

          "Now undo ×4 by dividing BOTH sides by 4.",

          "24 ÷ 4 = 6, so x = 6.",

        finalAnswer: "Subtract 6: 4x = 24. Divide by 4: x = 6. Check: 4(6)+6=30 ✅",

        coach: (msg) => {

          const t = msg.toLowerCase();

          if (t.includes("divide first")) return "Try undoing the +6 first. What’s the opposite of +6?";

          if (t.includes("subtract 6")) return "Good—what is 30 − 6?";

          if (t.includes("24")) return "Nice. Now you have 4x = 24. What’s next?";

          if (t.includes("divide by 4")) return "Yes—what is 24 ÷ 4?";

          if (t.includes("6")) return "If x = 6, do the check: 4(6)+6 = ?";

          return "What’s your first move: subtract 6 or divide by 4? (Pick one.)";

        }

      },


      {

        id: "eq3",

        title: "Multi-step Equations (Long Lesson)",

        difficulty: 4,

        time: "40–55 min",

        goal: "Solve equations like 2x + 5x − 4 = 24 by combining like terms first.",

        steps: [

          "Simplify each side: combine like terms (x terms with x terms, numbers with numbers).",

          "After simplifying, solve using the two-step equation method.",

          "Keep your work lined up to avoid mistakes.",

          "Check your answer in the ORIGINAL equation (not the simplified one only).",

        ],

        finalAnswer: "Combine: 2x+5x=7x → 7x−4=24. Add 4: 7x=28. Divide by 7: x=4. Check: 8+20−4=24 ✅",

        coach: (msg) => {

          const t = msg.toLowerCase();

          if (t.includes("7x - 4")) return "Great simplification. What should you do to undo −4?";

          if (t.includes("add 4")) return "Yes—what is 24 + 4?";

          if (t.includes("28")) return "Nice. Now you have 7x = 28. What next?";

          if (t.includes("4")) return "If x = 4, do the full check: 2(4)+5(4)−4 = ?";

    // APP STATE

    let current = lessons[0];

    let hintCount = 0;


    // INIT

    function init(){

      const sel = document.getElementById("lessonSelect");

      sel.innerHTML = lessons.map(l => `<option value="${l.id}">${l.title}</option>`).join("");

      loadLesson();

    }


    function loadLesson(){

      const id = document.getElementById("lessonSelect").value;

      current = lessons.find(l => l.id === id) || lessons[0];

      hintCount = 0;


      document.getElementById("goalBox").textContent = current.goal;

      document.getElementById("timePill").textContent = `Time: ${current.time}`;

      document.getElementById("skillsPill").textContent = `Skills: ${current.skills}`;


      const diff = "★".repeat(current.difficulty) + "☆".repeat(5-current.difficulty);

      document.getElementById("difficultyBadge").textContent = `Difficulty: ${diff}`;


      const steps = document.getElementById("stepsList");

      steps.innerHTML = current.steps.map(s => `<li>${s}</li>`).join("");


      document.getElementById("practiceBox").textContent = current.practice;


      resetChat(true);

      addAI(`New lesson loaded: "${current.title}". What’s your first step?`);

    }


    // CHAT UI

    function addAI(text){

      const chat = document.getElementById("chat");

      const div = document.createElement("div");

      div.className = "msg ai";

      div.textContent = text;

      chat.appendChild(div);

      chat.scrollTop = chat.scrollHeight;

    }

    function addKid(text){

      const chat = document.getElementById("chat");

      const div = document.createElement("div");

      div.className = "msg kid";

      div.textContent = text;

      chat.appendChild(div);

      chat.scrollTop = chat.scrollHeight;

    }


    // HINTS (B style: many hints, answer only after 6)

    function teacherHint(){

      const hints = current.hints || [];

      if (hints.length === 0){

        addAI("Hint: Tell me what step you tried so far, and I’ll guide you.");

        return;

      }


      const i = Math.min(hintCount, hints.length - 1);

      addAI("Hint " + (hintCount + 1) + ": " + hints[i]);

      hintCount++;


      if (hintCount >= teacherSettings.maxHintsBeforeAnswer){

        addAI('You’ve used lots of hints. If you want, type: SHOW ANSWER');

      }

    }


    function sendKid(){

      const input = document.getElementById("kidInput");

      const msg = input.value.trim();

      if(!msg) return;

      input.value = "";


      addKid(msg);


      const t = msg.toLowerCase();

      if (t === "show answer" || t === "show the answer"){

        if (hintCount >= teacherSettings.maxHintsBeforeAnswer){

          addAI("Final answer (with steps): " + (current.finalAnswer || "No finalAnswer found for this lesson."));

        } else {

          addAI(`Not yet—try a few more hints first. You’ve used ${hintCount}/${teacherSettings.maxHintsBeforeAnswer}.`);

        }

        return;

      }


      addAI(current.coach ? current.coach(msg) : "Tell me what you tried so far.");

    }


    function resetChat(silent){

      const chat = document.getElementById("chat");

      chat.innerHTML = "";

      hintCount = 0;

      if(!silent){

        addAI("Chat reset. Start by reading the practice question and telling me your first step.");

      }

    }


    init();
